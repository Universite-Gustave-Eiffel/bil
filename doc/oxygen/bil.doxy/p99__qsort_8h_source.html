80 #define P00_QSORT_BODY(SWAP) \
81 if (p00_n > RSIZE_MAX || p00_s > RSIZE_MAX) return EDOM; \
82 if (p00_n && (!p00_base || !p00_comp)) return EINVAL; \
83 do { \
84  register p99_seed *const seed = p99_seed_get(); \
85  /* Initialize a stack of states */ \
86  P99_CONSTANT(int, p00_tail, 20); \
87  /* provide a reasonable stack of states */ \
88  p00_qsort p00_a[(p99_arith_log2(p00_n) + 1)*p00_tail]; \
89  p00_qsort* p00_p = p00_a; \
90  P00_QPUSH(p00_p, 0, p00_n); \
91  while(!P00_QEMPTY(p00_a, p00_p)) { \
92  /* read the values from the stack */ \
93  P00_QTOP(p00_p, p00_bot, p00_top); \
94  register size_t const p00_len = p00_top - p00_bot; \
95  \
96  switch (p00_len) { \
97  /* special cases all fall through. */ \
98  case 2: \
99  if (P00_QCOMP(p00_bot, p00_bot + 1) > 0) \
100  SWAP(p00_B, p00_bot, p00_bot + 1); \
101  case 1: ; \
102  case 0: \
103  --p00_p; \
104  break; \
105  default: \
106  /* move the pivot to the bottom element */ \
107  { \
108  size_t p00_c = p99_rand(seed) % p00_len; \
109  if (p00_c) SWAP(p00_B, p00_bot, p00_bot + p00_c); \
110  } \
111  \
112  /* The running variables. The fact that p00_b is > p00_bot, \
113  guarantees progress. */ \
114  register size_t p00_b = p00_bot + 1; \
115  register size_t p00_t = p00_top; \
116  \
117  while (P99_LIKELY(p00_b < p00_t)) { \
118  /* Try to find two misplaced elements. */ \
119  while (!(P00_QCOMP(p00_bot, p00_b) < 0)) { \
120  ++p00_b; \
121  if (P99_UNLIKELY(p00_b == p00_t)) goto P00_RECURSE; \
122  } \
123  while (P00_QCOMP(p00_bot, p00_t - 1) < 0){ \
124  --p00_t; \
125  if (P99_UNLIKELY(p00_b == p00_t)) goto P00_RECURSE; \
126  } \
127  \
128  /* Two distinct misplaced elements are found. */ \
129  --p00_t; \
130  SWAP(p00_B, p00_b, p00_t); \
131  ++p00_b; \
132  } \
133  P00_RECURSE: \
134  if ((p00_b - p00_bot) > 1) { \
135  /* The pivot is always a maximal element in the lower part */ \
136  --p00_b; \
137  SWAP(p00_B, p00_bot, p00_b); \
138  /* Move all elements that compare equal adjacent */ \
139  register size_t const p00_bb = p00_b; \
140  for (register size_t p00_c = p00_bot; \
141  p00_c < p00_b;) { \
142  if (P00_QCOMP(p00_bb, p00_c) <= 0) { \
143  --p00_b; \
144  if (p00_c == p00_b) break; \
145  SWAP(p00_B, p00_c, p00_b); \
146  } else ++p00_c; \
147  } \
148  } \
149  /* Register the "recursive" calls into the two halfs. Most */ \
150  /* recursive calls are actually close to the bottom, so we */ \
151  /* rarely have to do the check if they are small enough. */ \
152  if (P99_UNLIKELY(p00_len >= p00_tail)) { \
153  size_t p00Mmax = p00_len*0.99; \
154  /* If the split isn't good enough we just try again */ \
155  if (P99_UNLIKELY(((p00_b - p00_bot) >= p00Mmax) \
156  || \
157  ((p00_top - p00_t) >= p00Mmax))) \
158  continue; \
159  } \
160  --p00_p; \
161  if ((p00_b - p00_bot) > 1) P00_QPUSH(p00_p, p00_bot, p00_b); \
162  if ((p00_top - p00_t) > 1) P00_QPUSH(p00_p, p00_t, p00_top); \
163  } \
164  } \
165  return 0; \
166  } while(false)
