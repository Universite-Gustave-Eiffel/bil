772 template<typename T>
773 struct PowExpr : BinaryExpr<T>
774 {
775  // Using declarations for data members of base class
776  using BinaryExpr<T>::val;
777  using BinaryExpr<T>::l;
778  using BinaryExpr<T>::r;
779 
780  T log_l;
781 
782  PowExpr(const T& v, const ExprPtr<T>& ll, const ExprPtr<T>& rr) : BinaryExpr<T>(v, ll, rr), log_l(log(ll->val)) {}

784  void propagate(const T& wprime) override
785  {
786  using U = VariableValueType<T>;
787  constexpr auto zero = U(0.0);
788  const auto lval = l->val;
789  const auto rval = r->val;
790  const auto aux = wprime * pow(lval, rval - 1);
791  l->propagate(aux * rval);
792  const auto auxr = lval == zero ? 0.0 : lval * log(lval); // since x*log(x) -> 0 as x -> 0
793  r->propagate(aux * auxr);
794  }

801  l->propagatex(aux * r);

803  r->propagatex(aux * auxr);
804  }
