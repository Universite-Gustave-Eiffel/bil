797 AUTODIFF_DEVICE_FUNC constexpr auto operator*(L&& l, R&& r)
798 {
799  // MULTIPLICATION EXPRESSION CASE: (-expr) * (-expr) => expr * expr
800  if constexpr (isNegExpr<L> && isNegExpr<R>)
801  return inner(l) * inner(r);
802  // // MULTIPLICATION EXPRESSION CASE: (1 / expr) * (1 / expr) => 1 / (expr * expr)
803  else if constexpr (isInvExpr<L> && isInvExpr<R>)
804  return inverse(inner(l) * inner(r));
805  // // MULTIPLICATION EXPRESSION CASE: expr * number => number * expr
806  else if constexpr (isExpr<L> && isArithmetic<R>)
807  return std::forward<R>(r) * std::forward<L>(l);
808  // // MULTIPLICATION EXPRESSION CASE: number * (-expr) => (-number) * expr
809  else if constexpr (isArithmetic<L> && isNegExpr<R>)
810  return (-l) * inner(r);
811  // // MULTIPLICATION EXPRESSION CASE: number * (number * expr) => (number * number) * expr
812  else if constexpr (isArithmetic<L> && isNumberDualMulExpr<R>)
813  return (l * left(r)) * right(r);
814  // MULTIPLICATION EXPRESSION CASE: number * dual => NumberDualMulExpr
815  else if constexpr (isArithmetic<L> && isDual<R>)
816  return NumberDualMulExpr<PreventExprRef<L>, PreventExprRef<R>>{ l, r };
817  // DEFAULT MULTIPLICATION EXPRESSION: expr * expr => MulExpr
818  else return MulExpr<PreventExprRef<L>, PreventExprRef<R>>{ l, r };
819 }

975 template<typename T, typename G, typename U>
976 AUTODIFF_DEVICE_FUNC constexpr void assign(Dual<T, G>& self, U&& other)
977 {
978  static_assert(isExpr<U> || isArithmetic<U>);
979 
980  // ASSIGN A NUMBER: self = number
981  if constexpr (isArithmetic<U>) {
982  self.val = other;
983  self.grad = Zero<G>();
984  }
985  // ASSIGN A DUAL NUMBER: self = dual
986  else if constexpr (isDual<U>) {
987  self.val = other.val;
988  self.grad = other.grad;
989  }
990  // ASSIGN A NUMBER-DUAL MULTIPLICATION EXPRESSION: self = number * dual
991  else if constexpr (isNumberDualMulExpr<U>) {
992  assign(self, other.r);
993  scale(self, other.l);
994  }
995  // ASSIGN A UNARY EXPRESSION: self = unaryexpr
996  else if constexpr (isUnaryExpr<U>) {
997  using Op = DualOpType<U>;
998  assign(self, other.r);
999  apply<Op>(self);
1000  }
1001  // ASSIGN AN ADDITION EXPRESSION: self = expr + expr
1002  else if constexpr (isAddExpr<U>) {
1003  assign(self, other.r);
1004  assignAdd(self, other.l);
1005  }
1006  // ASSIGN A MULTIPLICATION EXPRESSION: self = expr * expr
1007  else if constexpr (isMulExpr<U>) {
1008  assign(self, other.r);
1009  assignMul(self, other.l);
1010  }
1011  // ASSIGN A POWER EXPRESSION: self = pow(expr)
1012  else if constexpr (isPowExpr<U>) {
1013  assign(self, other.l);
1014  assignPow(self, other.r);
1015  }
1016  // ASSIGN A ATAN2 EXPRESSION: self = atan2(expr, expr)
1017  else if constexpr (isArcTan2Expr<U>) {
1018  assignArcTan2(self, other.l, other.r);
1019  }
1020 
1021  // ASSIGN A HYPOT2 EXPRESSION: self = hypot(expr, expr)
1022  else if constexpr (isHypot2Expr<U>) {
1023  assignHypot2(self, other.l, other.r);
1024  }
1025 
1026  // ASSIGN A HYPOT3 EXPRESSION: self = hypot(expr, expr)
1027  else if constexpr (isHypot3Expr<U>) {
