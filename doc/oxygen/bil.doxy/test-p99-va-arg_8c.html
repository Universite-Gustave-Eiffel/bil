TYPE P99_PASTE3(p00_, gcd_, SUFF)(TYPE a, TYPE b) { \
if (!a) return b; \
if (b) \
for (TYPE mod = b % a; mod; mod = b % a) { \
b = a; \
a = mod; \
} \
return a; \
} \
TYPE P99_FSYMB(P99_PASTE3(p99_, gcd_, SUFF))(size_t number, TYPE const*const arr) { \
if (number == 1) return arr[0]; \

for (size_t i = 2; i < number; ++i) \

return ret; \
} \
