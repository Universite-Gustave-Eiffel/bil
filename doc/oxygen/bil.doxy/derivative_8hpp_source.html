91 template<typename Var, typename... Vars, typename T>
92 AUTODIFF_DEVICE_FUNC auto seed(const Wrt<Var&, Vars&...>& wrt, T&& seedval)
93 {
94  constexpr static auto N = Order<Var>;
95  constexpr static auto size = 1 + sizeof...(Vars);
96  static_assert(size <= N, "It is not possible to compute higher-order derivatives with order greater than that of the autodiff number (e.g., using wrt(x, x, y, z) will fail if the autodiff numbers in use have order below 4).");
97  For<N>([&](auto i) constexpr {
98  if constexpr (i.index < size)
99  seed<i.index + 1>(std::get<i>(wrt.args), seedval);
100  else
101  seed<i.index + 1>(std::get<size - 1>(wrt.args), seedval); // use the last variable in the wrt list as the variable for which the remaining higher-order derivatives are calculated (e.g., derivatives(f, wrt(x), at(x)) will produce [f0, fx, fxx, fxxx, fxxxx] when x is a 4th order dual number).
102  });
103 }

117 template<typename... Args, typename... Vecs>
118 AUTODIFF_DEVICE_FUNC auto seed(const At<Args...>& at, const Along<Vecs...>& along)
119 {
120  static_assert(sizeof...(Args) == sizeof...(Vecs));
121 
122  ForEach(at.args, along.args, [&](auto& arg, auto&& dir) constexpr {
123  if constexpr (isVector<decltype(arg)>) {
124  static_assert(isVector<decltype(dir)>);
125  assert(arg.size() == dir.size());
126  for(auto i = 0; i < dir.size(); ++i)
127  seed<1>(arg[i], dir[i]);
128  }
129  else seed<1>(arg, dir);
130  });
131 }

200 template<size_t order = 1, typename Vec, Requires<isVector<Vec>> = true>
201 AUTODIFF_DEVICE_FUNC auto derivative(const Vec& u)
202 {
203  size_t len = u.size(); // the length of the vector containing dual/real numbers
204  using NumType = decltype(u[0]); // get the type of the dual/real number
205  using T = NumericType<NumType>; // get the numeric/floating point type of the dual/real number
206  using Res = VectorReplaceValueType<Vec, T>; // get the type of the vector containing numeric values instead of dual/real numbers (e.g., vector<real> becomes vector<double>, VectorXdual becomes VectorXd, etc.)
207  Res res(len); // create an array to store the derivatives stored inside the dual/real number
208  for(auto i = 0; i < len; ++i)
209  res[i] = derivative<order>(u[i]); // get the derivative of given order from i-th dual/real number
210  return res;
211 }

221 template<typename Result>
222 AUTODIFF_DEVICE_FUNC auto derivatives(const Result& result)
223 {
224 #ifndef __CUDA_ARCH__
225  if constexpr (isVector<Result>) // check if the argument is a vector container of dual/real numbers
226  {
227  size_t len = result.size(); // the length of the vector containing dual/real numbers
228  using NumType = decltype(result[0]); // get the type of the dual/real number
229  using T = NumericType<NumType>; // get the numeric/floating point type of the dual/real number
230  using Vec = VectorReplaceValueType<Result, T>; // get the type of the vector containing numeric values instead of dual/real numbers (e.g., vector<real> becomes vector<double>, VectorXdual becomes VectorXd, etc.)
231  constexpr auto N = Order<NumType>; // the order of the dual/real number
232  std::array<Vec, N + 1> values; // create an array to store the derivatives stored inside the dual/real number
233  For<N + 1>([&](auto i) constexpr {
234  values[i].resize(len);
235  for(auto j = 0U; j < len; ++j)
236  values[i][j] = derivative<i>(result[j]); // get the ith derivative of the jth dual/real number
237  });
238  return values;
239  }
240  else // result is then just a dual/real number
241 #endif
242  {
243  using T = NumericType<Result>; // get the numeric/floating point type of the dual/real result number
244  constexpr auto N = Order<Result>; // the order of the dual/real result number
245  std::array<T, N + 1> values; // create an array to store the derivatives stored inside the dual/real number
246  For<N + 1>([&](auto i) constexpr {
247  values[i] = derivative<i>(result);
248  });
249  return values;
250  }
251 }
