154 p99_inline
155 int64_t p99_uf_union(p99_uf* uf, uint64_t left, uint64_t right) {
156  int64_t root = -1;
157  if (uf && (uf->size > left) && (uf->size > right)) {
158  root = p00_uf_findCompress(left, uf->tab);
159  if (root >= 0) {
160  int64_t rright = p00_uf_find(right, uf->tab);
161  if (rright >= 0) {
162  // use the new root to mark all elements of the right
163  p00_uf_compress(right, uf->tab, root);
164  // now also link the old root on the right to the new one
165  p00_uf_union(root, rright, uf->tab);
166  } else
167  root = rright;
168  }
169  }
170  return root;
171 }
