107  register p99_bitset const p00_b0 = p00_bits0(p00_s2l, p00_s2);

111 
112  register p99_bitset p00_ret = p00_set_bit(p00_c);
113  switch (p00_c & p00_bitset_bit3) {
114  default: p00_ret &= p00_b0; break;
115  case p00_bitset_bit1: p00_ret &= p00_b1; break;
116  case p00_bitset_bit2: p00_ret &= p00_b2; break;
117  case p00_bitset_bit3: p00_ret &= p00_b3; break;
118  }
119  return p00_ret;
120 }
