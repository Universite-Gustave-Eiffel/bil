71 /*
72  * *************************************************
73  * Global data structures used in LU factorization
74  * *************************************************
75  *
76  * nsuper: number of supernodes = nsuper+1, numbered between 0 and nsuper.
77  *
78  * (supno, xsup, xsup_end):
79  * supno[i] is the supernode number to which column i belongs;
80  * xsup[s] points to the first column of supernode s;
81  * xsup_end[s] points to one past the last column of supernode s.
82  * Example: supno 0 1 2 2 3 3 3 4 4 4 4 4 (n=12)
83  * xsup 0 1 2 4 7
84  * xsup_end 1 2 4 7 12
85  * Note: dfs will be performed on supernode rep. relative to the new
86  * row pivoting ordering
87  *
88  * (lsub, xlsub, xlsub_end):
89  * lsub[*] contains the compressed subscripts of the supernodes;
90  * xlsub[j] points to the starting location of the j-th column in
91  * lsub[*];
92  * xlsub_end[j] points to one past the ending location of the j-th
93  * column in lsub[*].
94  * Storage: original row subscripts in A.
95  *
96  * During the course of sparse LU factorization, we also use
97  * (lsub, xlsub, xlsub_end, xprune) to represent symmetrically
98  * pruned graph. Contention will occur when one processor is
99  * performing DFS on supernode S, while another processor is pruning
100  * supernode S. We use the following data structure to deal with
101  * this problem. Suppose each supernode contains columns {s,s+1,...,t},
102  * with first column s and last column t.
103  *
104  * (1) if t > s, only the subscript sets for column s and column t
105  * are stored. Column t represents pruned adjacency structure.
106  *
107  * --------------------------------------------
108  * lsub[*] ... | col s | col t | ...
109  * --------------------------------------------
110  * ^ ^ ^
111  * xlsub[s] xlsub_end[s] xlsub_end[s+1]
112  * xlsub[s+1] :
113  * : :
114  * : xlsub_end[t]
115  * xlsub[t] xprune[t]
116  * xprune[s]
117  *
118  * (2) if t == s, i.e., a singleton supernode, the subscript set
119  * is stored twice:
120  *
121  * --------------------------------------
122  * lsub[*] ... | s | s | ...
123  * --------------------------------------
124  * ^ ^ ^
125  * xlsub[s] xlsub_end[s] xprune[s]
126  *
127  * There are two subscript sets for each supernode, the last column
128  * structures (for pruning) will be removed after the numerical LU
129  * factorization phase:
130  * o lsub[j], j = xlsub[s], ..., xlsub_end[s]-1
131  * is the structure of column s (i.e. structure of this supernode).
132  * It is used for the storage of numerical values.
133  * o lsub[j], j = xlsub[t], ..., xlsub_end[t]-1
134  * is the structure of the last column t of this supernode.
135  * It is for the purpose of symmetric pruning. Therefore, the
136  * structural subscripts can be rearranged without making physical
137  * interchanges among the numerical values.
138  *
139  * DFS will traverse the first subscript set if the supernode
140  * has not been pruned; otherwise it will traverse the second
141  * subscript list, i.e., the part of the pruned graph.
142  *
143  * (lusup, xlusup, xlusup_end):
144  * lusup[*] contains the numerical values of the supernodes;
145  * xlusup[j] points to the starting location of the j-th column in
146  * storage vector lusup[*];
147  * xlusup_end[j] points to one past the ending location of the j-th
148  * column in lusup[*].
149  * Each supernode is stored in column-major, consistent with Fortran
150  * two-dimensional array storage.
151  *
152  * (ucol, usub, xusub, xusub_end):
153  * ucol[*] stores the numerical values of the U-columns above the
154  * supernodes.
155  * usub[k] stores the row subscripts of nonzeros ucol[k];
156  * xusub[j] points to the starting location of column j in ucol/usub[];
157  * xusub_end[j] points to one past the ending location column j in
158  * ucol/usub[].
159  * Storage: new row subscripts; that is indexed intp PA.
160  *
161  */
162 typedef struct {
163  int_t *xsup; /* supernode and column mapping */
164  int_t *xsup_end;
165  int_t *supno;
166  int_t *lsub; /* compressed L subscripts */
167  int_t *xlsub;
168  int_t *xlsub_end;
169  double *lusup; /* L supernodes */
170  int_t *xlusup;
171  int_t *xlusup_end;
172  double *ucol; /* U columns */
173  int_t *usub;
174  int_t *xusub;
175  int_t *xusub_end;
176  int_t nsuper; /* current supernode number */
177  int_t nextl; /* next position in lsub[] */
178  int_t nextu; /* next position in usub[]/ucol[] */
179  int_t nextlu; /* next position in lusup[] */
180  int_t nzlmax; /* current max size of lsub[] */
181  int_t nzumax; /* " " " ucol[] */
182  int_t nzlumax; /* " " " lusup[] */
183  /* ---------------------------------------------------------------
184  * Memory managemant for L supernodes
185  */
186  int_t *map_in_sup; /* size n+1 - the address offset of each column
187  * in lusup[*], which is divided into regions
188  * by the supernodes of Householder matrix H.
189  * If column k starts a supernode in H,
190  * map_in_sup[k] is the next open position in
191  * lusup[*]; otherwise map_in_sup[k] gives the
192  * offset (negative) to the leading column
193  * of the supernode in H.
194  */
195  int_t dynamic_snode_bound;
196  /* --------------------------------------------------------------- */
197 } GlobalLU_t;
