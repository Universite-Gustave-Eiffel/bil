88 /* The tag part of a p99_tp ideally would be unique during the whole
89  run of a program instance. Since we might not have more than 32 bit
90  for this, this will not be possible in general, but we try to get
91  as close as possible to that.
92 
93  The problem should generally only arise for several accesses to the
94  same memory that are close in time. Particularly problematic are
95  such usages that go across free and re-malloc boundaries, because
96  two instances of p99_tp that are realized on the same address could
97  look the same if by coincidence their tags are the same.
98 
99  Therefore we compose each tag from two parts. One is a thread
100  specific counter (p00_tp_tick), and the other is a global counter
101  (p00_tp_tack) that is only consulted, when the thread counter
102  wraps.
103 
104  If uintptr_t is 64 bit wide this should be safe. Each of the type
105  of counters only would wrap after 4 billion increments. In
106  particular we only could have a duplicate tag after 4 billion
107  events that consist in the creation of a new thread or in
108  p00_tp_tick wrapping for certain threads.
109 
110  */
111 P99_DECLARE_THREAD_LOCAL(uintptr_t volatile, p00_tp_tick);
112 P99_WEAK(p00_tp_tack) _Atomic(uintptr_t) volatile p00_tp_tack;
113 
114 p99_inline
115 uintptr_t p00_tp_tick_get(void) {
116  enum { p00_bits = sizeof(uintptr_t)*CHAR_BIT/2, };
117  register uintptr_t const p00_mask = (UINTPTR_MAX >> p00_bits);
118  register uintptr_t volatile*const p00_ret = &P99_THREAD_LOCAL(p00_tp_tick);
119  if (P99_UNLIKELY(!(*p00_ret & p00_mask))) {
120  uintptr_t p00_tack = 0;
121  while (!p00_tack) {
122  p00_tack = atomic_fetch_add_explicit(&p00_tp_tack, 1u, memory_order_acq_rel);
123  p00_tack &= p00_mask;
124  }
125  *p00_ret = (p00_tack << p00_bits);
126  }
127  return ++(*p00_ret);
128 }

192 p99_inline
193 p00_tp_glue p00_tp_get(register p99_tp volatile*const p00_tp) {
194  register p00_tp_glue p00_ret
195  = P99_LIKELY(p00_tp)
196  ? atomic_load_explicit(&p00_tp->p00_val, memory_order_consume)
197  : (p00_tp_glue)P00_TP_GLUE_INITIALIZER((void*)0, p00_tp_tick_get());
198  if (p00_tp && P99_UNLIKELY(!p00_tp_i2i(p00_ret))) {
199  /* Only store it in addressable memory if we can't avoid it. */
200  p00_tp_glue p00_ter = p00_ret;
201  register p00_tp_glue p00_rep = P00_TP_GLUE_INITIALIZER(p00_tp->p00_init, p00_tp_tick_get());
202  if (p00_tp_cmpxchg(&p00_tp->p00_val, &p00_ter, p00_rep))
203  p00_ret = p00_rep;
204  else
205  p00_ret = p00_ter;
206  }
207  return p00_ret;
208 }

213 p99_inline
214 p00_tp_glue p99_tp_xchg(p99_tp volatile* p00_tp, void* p00_val) {
215  p00_tp_glue p00_ret
216  = P99_LIKELY(p00_tp)
217  ? atomic_load_explicit(&p00_tp->p00_val, memory_order_consume)
218  : (p00_tp_glue)P00_TP_GLUE_INITIALIZER((void*)0, p00_tp_tick_get());
219  if (P99_LIKELY(p00_tp)) {
220  register p00_tp_glue p00_rep = P00_TP_GLUE_INITIALIZER(p00_val, p00_tp_tick_get());
221  while (!p00_tp_cmpxchg(&p00_tp->p00_val, &p00_ret, p00_rep));
222  /* if this p99_tp has not been used before, return the value that
223  p99_tp_get would have returned. */
224  if (P99_UNLIKELY(!p00_tp_i2i(p00_ret))) {
225  p00_ret = (p00_tp_glue)P00_TP_GLUE_INITIALIZER(p00_tp->p00_init, p00_tp_tick_get());
226  }
227  }
228  return p00_ret;
229 }
