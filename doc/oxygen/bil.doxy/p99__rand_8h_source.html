42 p99_inline
43 uint32_t p00_xorshift(p00_seed160 * p00_s) {
44  /* Use the 6th element as the running index, and compute the
45  necessary derived indices mod 5 */
46  ++(*p00_s)[5];
47  register uint32_t const p00_0 = ((*p00_s)[5]) % 5;
48  register uint32_t const p00_4 = p00_0 == 0 ? 4 : p00_0-1;
49  register uint32_t const p00_2 = (p00_0 == 4) ? 1 : ((p00_0 == 3) ? 0 : p00_0+2);
50 
51  /* Load the part of the state variable that we need. */
52  register uint32_t const p00_x = (*p00_s)[p00_0];
53  register uint32_t const p00_y = (*p00_s)[p00_2];
54  register uint32_t const p00_v = (*p00_s)[p00_4];
55 
56  register uint32_t t = (p00_x^(p00_x>>7));
57  t = (p00_v^(p00_v<<6))^(t^(t<<13));
58 
59  /* Change just one place in the state variables. */
60  (*p00_s)[p00_4] = t;
61 
62  /* Multiply with an odd number */
63  return (2*p00_y + 1) * t;
64 }

212 uint32_t p00_bitpack(void const* p00_p) {
213  uintptr_t p00_u = (uintptr_t)p00_p;
214 #if UINTPTR_MAX == UINT32_MAX
215  return p00_u;
216 #else
217  uint32_t p00_r = 0;
218  do {
219  p00_r ^= p00_u;
220  p00_u >>= (sizeof(uint32_t)*CHAR_BIT);
221  } while (p00_u);
222  return p00_r;
223 #endif
224 }

227 P99_WEAK(p00_rand_init)
228 void p00_rand_init(void* p00_p) {
229  p99_seed * p00_s = p00_p
230  ? p00_p
231  : &P99_THREAD_LOCAL(p00_seed).p00_seed;
232  /* local to this initialization call, stack address */
233  uint32_t p00_0 = p00_bitpack(&p00_s);
234  /* unique to this thread, heap address */
235  uint32_t p00_1 = p00_bitpack(p00_p);
236  struct timespec p00_ts;
237  timespec_get(&p00_ts, TIME_UTC);
238  uint32_t p00_2 = p00_ts.tv_sec;
239  uint32_t p00_3 = p00_ts.tv_nsec;
240 #ifdef TIME_MONOTONIC
241  struct timespec p00_tm;
242  timespec_get(&p00_tm, TIME_MONOTONIC);
243  uint32_t p00_4 = p00_tm.tv_nsec;
244 #else
245  uint32_t p00_4 = p00_bitpack(__func__);
246 #endif
247  /* index, unique to this thread */
248  uint32_t p00_ind = p00_0 ^ p00_1 ^ p00_2 ^ p00_3 ^ p00_4;
249  p00_seed160 p00_st = {
250  p00_0,
251  p00_1,
252  p00_2,
253  p00_3,
254  p00_4,
255  p00_ind,
256  };
257  /* mix things up a bit */
258  for (unsigned p00_i = 0; p00_i < 32; ++p00_i) p00_xorshift(&p00_st);
259  /* Now create two different state vectors that are de-phased. We
260  can be in any state of the xorshift generator, so there are
261  2^160 different initializations. */
262  for (unsigned p00_j = 0; p00_j < 2; ++p00_j) {
263  for (unsigned p00_i = 0; p00_i < p00_seed160_len; ++p00_i)
264  (*p00_s)[p00_j][p00_i] = p00_xorshift(&p00_st);
265  }
266 }

330 P99_DEFARG_DOCU(p99_rand)
331 p99_inline
332 uint64_t p99_rand(register p99_seed * p00_seed) {
333  uint32_t p00_0 = p00_xorshift(&(*p00_seed)[0]);
334  uint64_t p00_1 = p00_xorshift(&(*p00_seed)[1]);
335  uint64_t p00_0r = p00_0 % p00_bigprime_len;
336  uint64_t p00_0d = p00_0 / p00_bigprime_len;
337  /* Use part of the bits to choose a big number */
338  p00_1 *= p00_bigprime[p00_0r];
339  /* Use the rest of the bits to add more randomness */
340  return (p00_0d ^ p00_1);
341 }

364 double p99_drand(register p99_seed * p00_seed) {
365  enum {
366  p00_s = (DBL_MANT_DIG > 64 ? 64 : DBL_MANT_DIG),
367  p00_0 = 64 - p00_s,
368  p00_m = (1 << p00_0),
369  };
370  double const p00_imax = 1.0 / (UINT64_C(1) << p00_s);
371  uint64_t p00_r = p99_rand(p00_seed);
372  double p00_1 = ((p00_r >> p00_0)^(p00_r & p00_m)) * p00_imax;
373  return p00_1;
374 }
