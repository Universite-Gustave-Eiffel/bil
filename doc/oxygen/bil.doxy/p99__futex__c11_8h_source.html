71 /* Supposes that the lock on the mutex is already taken and that
72  * p00_wmin <= p00_wmax. Returns min(p00_win, p00_wok) where p00_wok
73  * is the number of threads that have been woken up. */
74 p99_inline
75 unsigned p00_futex_wakeup(p99_futex volatile* p00_fut,
76  unsigned p00_wmin, unsigned p00_wmax) {
77  assert(p00_wmin <= p00_wmax);
78  if (p00_wmax && p00_fut->p99_waiting) {
79  if (p00_wmax > p00_fut->p99_waiting) p00_wmax = p00_fut->p99_waiting;
80  if (p00_wmax > 1u) cnd_broadcast((cnd_t*)&p00_fut->p99_cnd);
81  else cnd_signal((cnd_t*)&p00_fut->p99_cnd);
82  p00_fut->p99_waiting -= p00_wmax;
83  p00_fut->p99_awaking += p00_wmax;
84  return p00_wmin;
85  } else {
86  return 0;
87  }
88 }

97 P99_WEAK(p99_futex_wakeup)
98 void p99_futex_wakeup(p99_futex volatile* p00_fut,
99  unsigned p00_wmin, unsigned p00_wmax) {
100  if (p00_wmax < p00_wmin) p00_wmax = p00_wmin;
101  if (p00_wmax) do {
102  unsigned p00_wok = 0;
103  P99_MUTUAL_EXCLUDE(*(mtx_t*)&p00_fut->p99_mut) {
104  p00_wok = p00_futex_wakeup(p00_fut, p00_wmin, p00_wmax);
105  }
106  p00_wmax -= p00_wok;
107  p00_wmin -= p00_wok;
108  } while (p00_wmin);
109 }

111 p99_inline
112 void p00_futex_wait(p99_futex volatile* p00_fut) {
113  ++p00_fut->p99_waiting;
114  /* This loop captures spurious wakeups as they may happen for
115  cnd_wait. */
116  do {
117  cnd_wait((cnd_t*)&p00_fut->p99_cnd, (mtx_t*)&p00_fut->p99_mut);
118  } while (!p00_fut->p99_awaking);
119  --p00_fut->p99_awaking;
120 }

128 P99_WEAK(p99_futex_add)
129 unsigned p99_futex_add(p99_futex volatile* p00_fut, unsigned p00_hmuch,
130  unsigned p00_cstart, unsigned p00_clen,
131  unsigned p00_wmin, unsigned p00_wmax) {
132  unsigned p00_ret = 0;
133  if (p00_wmax < p00_wmin) p00_wmax = p00_wmin;
134  P99_MUTUAL_EXCLUDE(*(mtx_t*)&p00_fut->p99_mut) {
135  p00_ret = p00_fut->p99_cnt;
136  register unsigned const p00_des = p00_ret + p00_hmuch;
137  p00_fut->p99_cnt = p00_des;
138  if (p00_clen && P99_IN_RANGE(p00_des, p00_cstart, p00_clen)) {
139  P00_FUTEX_WAKEUP(p00_fut, p00_wmin, p00_wmax);
140  } else {
141  p00_wmin = 0;
142  }
143  }
144  /* If we haven't woken enough threads, we have to re-acquire the
145  mutex and loop until enough threads will enter the futex wait. */
146  if (p00_wmin) p99_futex_wakeup(p00_fut, p00_wmin, p00_wmax);
147  return p00_ret;
148 }
