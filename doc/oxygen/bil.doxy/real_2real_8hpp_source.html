667  Real<N, T> res;
668  res[0] = atan2(c, x[0]);
669  if constexpr(N > 0) {
670  Real<N - 1, T> xprime;
671  For<1, N + 1>([&](auto i) constexpr {
672  xprime[i - 1] = x[i];
673  });
674  Real<N - 1, T> aux(x);
675  aux = xprime * (-c / (c * c + aux * aux));
676  For<1, N + 1>([&](auto i) constexpr {
677  res[i] = aux[i - 1];
678  });
679  }
680  return res;
681 }

762 {

779  constexpr auto c = BinomialCoefficient<i.index - 1, j.index>;
