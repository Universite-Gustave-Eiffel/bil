246 /* This is an implementation to bootstrap the thread specific
247  code. Once initialization functionalities that are better suited
248  for application code are defined elsewhere. */
249 p99_inline
250 void p00_tss_init(p99_tss * p00_key) {
251  if (P99_UNLIKELY(!p00_key->p00_done)) {
252  P99_SPIN_EXCLUDE(&p00_key->p00_flg) {
253  if (!p00_key->p00_done) {
254  int p00_ret = tss_create(&P99_ENCP(p00_key), p00_key->p99_dtor);
255  if (p00_ret) {
256  errno = p00_ret;
257  perror("can't create thread specific key");
258  abort();
259  }
260  p00_key->p00_done = true;
261  }
262  }
263  }
264 }

305 p99_inline
306 P99_WARN_UNUSED_RESULT
307 int p99_tss_set(p99_tss * p00_key, void *p00_val) {
308  int p00_ret = thrd_success;
309  void * p00_vol = p99_tss_get(p00_key);
310  if (p00_val != p00_vol) {
311  p00_ret = tss_set(P99_ENCP(p00_key), p00_val);
312  if (p00_ret == thrd_success && p00_vol && p00_key->p99_dtor) {
313  p00_key->p99_dtor(p00_vol);
314  }
315  }
316  return p00_ret;
317 }

434 void* p99_tss_get_alloc(p99_tss * p00_key, size_t p00_size) {
435  void * p00_ret = p99_tss_get(p00_key);
436  if (P99_UNLIKELY(!p00_ret))
437  if (p00_size) {
438  p00_ret = calloc(1, p00_size);

440  free(p00_ret);
441  p00_ret = 0;
442  }
