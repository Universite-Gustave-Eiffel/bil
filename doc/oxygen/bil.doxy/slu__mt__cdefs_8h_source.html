72 /*
73  * *************************************************
74  * Global data structures used in LU factorization
75  * *************************************************
76  *
77  * nsuper: number of supernodes = nsuper+1, numbered between 0 and nsuper.
78  *
79  * (supno, xsup, xsup_end):
80  * supno[i] is the supernode number to which column i belongs;
81  * xsup[s] points to the first column of supernode s;
82  * xsup_end[s] points to one past the last column of supernode s.
83  * Example: supno 0 1 2 2 3 3 3 4 4 4 4 4 (n=12)
84  * xsup 0 1 2 4 7
85  * xsup_end 1 2 4 7 12
86  * Note: dfs will be performed on supernode rep. relative to the new
87  * row pivoting ordering
88  *
89  * (lsub, xlsub, xlsub_end):
90  * lsub[*] contains the compressed subscripts of the supernodes;
91  * xlsub[j] points to the starting location of the j-th column in
92  * lsub[*];
93  * xlsub_end[j] points to one past the ending location of the j-th
94  * column in lsub[*].
95  * Storage: original row subscripts in A.
96  *
97  * During the course of sparse LU factorization, we also use
98  * (lsub, xlsub, xlsub_end, xprune) to represent symmetrically
99  * pruned graph. Contention will occur when one processor is
100  * performing DFS on supernode S, while another processor is pruning
101  * supernode S. We use the following data structure to deal with
102  * this problem. Suppose each supernode contains columns {s,s+1,...,t},
103  * with first column s and last column t.
104  *
105  * (1) if t > s, only the subscript sets for column s and column t
106  * are stored. Column t represents pruned adjacency structure.
107  *
108  * --------------------------------------------
109  * lsub[*] ... | col s | col t | ...
110  * --------------------------------------------
111  * ^ ^ ^
112  * xlsub[s] xlsub_end[s] xlsub_end[s+1]
113  * xlsub[s+1] :
114  * : :
115  * : xlsub_end[t]
116  * xlsub[t] xprune[t]
117  * xprune[s]
118  *
119  * (2) if t == s, i.e., a singleton supernode, the subscript set
120  * is stored twice:
121  *
122  * --------------------------------------
123  * lsub[*] ... | s | s | ...
124  * --------------------------------------
125  * ^ ^ ^
126  * xlsub[s] xlsub_end[s] xprune[s]
127  *
128  * There are two subscript sets for each supernode, the last column
129  * structures (for pruning) will be removed after the numerical LU
130  * factorization phase:
131  * o lsub[j], j = xlsub[s], ..., xlsub_end[s]-1
132  * is the structure of column s (i.e. structure of this supernode).
133  * It is used for the storage of numerical values.
134  * o lsub[j], j = xlsub[t], ..., xlsub_end[t]-1
135  * is the structure of the last column t of this supernode.
136  * It is for the purpose of symmetric pruning. Therefore, the
137  * structural subscripts can be rearranged without making physical
138  * interchanges among the numerical values.
139  *
140  * DFS will traverse the first subscript set if the supernode
141  * has not been pruned; otherwise it will traverse the second
142  * subscript list, i.e., the part of the pruned graph.
143  *
144  * (lusup, xlusup, xlusup_end):
145  * lusup[*] contains the numerical values of the supernodes;
146  * xlusup[j] points to the starting location of the j-th column in
147  * storage vector lusup[*];
148  * xlusup_end[j] points to one past the ending location of the j-th
149  * column in lusup[*].
150  * Each supernode is stored in column-major, consistent with Fortran
151  * two-dimensional array storage.
152  *
153  * (ucol, usub, xusub, xusub_end):
154  * ucol[*] stores the numerical values of the U-columns above the
155  * supernodes.
156  * usub[k] stores the row subscripts of nonzeros ucol[k];
157  * xusub[j] points to the starting location of column j in ucol/usub[];
158  * xusub_end[j] points to one past the ending location column j in
159  * ucol/usub[].
160  * Storage: new row subscripts; that is indexed intp PA.
161  *
162  */
163 typedef struct {
164  int_t *xsup; /* supernode and column mapping */
165  int_t *xsup_end;
166  int_t *supno;
167  int_t *lsub; /* compressed L subscripts */
168  int_t *xlsub;
169  int_t *xlsub_end;
170  complex *lusup; /* L supernodes */
171  int_t *xlusup;
172  int_t *xlusup_end;
173  complex *ucol; /* U columns */
174  int_t *usub;
175  int_t *xusub;
176  int_t *xusub_end;
177  int_t nsuper; /* current supernode number */
178  int_t nextl; /* next position in lsub[] */
179  int_t nextu; /* next position in usub[]/ucol[] */
180  int_t nextlu; /* next position in lusup[] */
181  int_t nzlmax; /* current max size of lsub[] */
182  int_t nzumax; /* " " " ucol[] */
183  int_t nzlumax; /* " " " lusup[] */
184  /* ---------------------------------------------------------------
185  * Memory managemant for L supernodes
186  */
187  int_t *map_in_sup; /* size n+1 - the address offset of each column
188  * in lusup[*], which is divided into regions
189  * by the supernodes of Householder matrix H.
190  * If column k starts a supernode in H,
191  * map_in_sup[k] is the next open position in
192  * lusup[*]; otherwise map_in_sup[k] gives the
193  * offset (negative) to the leading column
194  * of the supernode in H.
195  */
196  int_t dynamic_snode_bound;
197  /* --------------------------------------------------------------- */
198 } GlobalLU_t;
