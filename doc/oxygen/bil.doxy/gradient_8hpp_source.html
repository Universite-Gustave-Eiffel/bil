60 // Loop through each variable in a wrt list and apply a function f(i, x) that
61 // accepts an index i and the variable x[i], where i is the global index of the
62 // variable in the list.
63 template<typename Function, typename... Vars>
64 constexpr auto ForEachWrtVar(const Wrt<Vars...>& wrt, Function&& f)
65 {
66  auto i = 0; // the current index of the variable in the wrt list
67  ForEach(wrt.args, [&](auto& item) constexpr
68  {
69  using T = decltype(item);
70  static_assert(isVector<T> || Order<T> > 0, "Expecting a wrt list with either vectors or individual autodiff numbers.");
71  if constexpr (isVector<T>) {
72  for(auto j = 0; j < item.size(); ++j)
73  // call given f with current index and variable from item (a vector)
74  if constexpr (detail::has_operator_bracket<T>()) {
75  f(i++, item[j]);
76  } else {
77  f(i++, item(j));
78  }
79  }
80  else f(i++, item); // call given f with current index and variable from item (a number, not a vector)
81  });
82 }

85 template<typename Fun, typename... Vars, typename... Args, typename Y, typename G>
86 void gradient(const Fun& f, const Wrt<Vars...>& wrt, const At<Args...>& at, Y& u, G& g)
87 {
88  static_assert(sizeof...(Vars) >= 1);
89  static_assert(sizeof...(Args) >= 1);
90 
91  const size_t n = wrt_total_length(wrt);
92 
93  g.resize(n);
94 
95  if(n == 0) return;
96 
97  ForEachWrtVar(wrt, [&](auto&& i, auto&& xi) constexpr
98  {
99  static_assert(!isConst<decltype(xi)>, "Expecting non-const autodiff numbers in wrt list because these need to be seeded, and thus altered!");
100  u = eval(f, at, detail::wrt(xi)); // evaluate u with xi seeded so that du/dxi is also computed
101  g[i] = derivative<1>(u);
102  });
103 
104 }

127 template<typename Fun, typename... Vars, typename... Args, typename Y, typename Jac>
128 void jacobian(const Fun& f, const Wrt<Vars...>& wrt, const At<Args...>& at, Y& F, Jac& J)
129 {
130  static_assert(sizeof...(Vars) >= 1);
131  static_assert(sizeof...(Args) >= 1);
132 
133  size_t n = wrt_total_length(wrt);
134  size_t m = 0;
135 
136  ForEachWrtVar(wrt, [&](auto&& i, auto&& xi) constexpr {
137  static_assert(!isConst<decltype(xi)>, "Expecting non-const autodiff numbers in wrt list because these need to be seeded, and thus altered!");
138  F = eval(f, at, detail::wrt(xi)); // evaluate F with xi seeded so that dF/dxi is also computed
139  if(m == 0) { m = F.size(); J.resize(m, n); };
140  for(size_t row = 0; row < m; ++row)
141  J(row, i) = derivative<1>(F[row]);
142  });
143 }

173 template<typename Fun, typename... Vars, typename... Args, typename U, typename G, typename H>
174 void hessian(const Fun& f, const Wrt<Vars...>& wrt, const At<Args...>& at, U& u, G& g, H& h)
175 {
176  static_assert(sizeof...(Vars) >= 1);
177  static_assert(sizeof...(Args) >= 1);
178 
179  size_t n = wrt_total_length(wrt);
180 
181  g.resize(n);
182  h.resize(n, n);
183 
184  ForEachWrtVar(wrt, [&](auto&& i, auto&& xi) constexpr {
185  ForEachWrtVar(wrt, [&](auto&& j, auto&& xj) constexpr
186  {
187  static_assert(!isConst<decltype(xi)> && !isConst<decltype(xj)>, "Expecting non-const autodiff numbers in wrt list because these need to be seeded, and thus altered!");
188  if(j >= i) { // this take advantage of the fact the Hessian matrix is symmetric
189  u = eval(f, at, detail::wrt(xi, xj)); // evaluate u with xi and xj seeded to produce u0, du/dxi, d2u/dxidxj
190  g[i] = derivative<1>(u); // get du/dxi from u
191  h(i, j) = h(j, i) = derivative<2>(u); // get d2u/dxidxj from u
192  }
193  });
194  });
195 }
