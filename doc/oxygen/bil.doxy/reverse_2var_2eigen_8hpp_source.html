121 template<typename T, typename X>
122 auto gradient(const Variable<T>& y, Eigen::DenseBase<X>& x)
123 {
124  using U = VariableValueType<T>;
125 
126  using ScalarX = typename X::Scalar;
127  static_assert(isVariable<ScalarX>, "Argument x is not a vector with Variable<T> (aka var) objects..");
128 
129  constexpr auto isVec = X::IsVectorAtCompileTime;
130  static_assert(isVec, "Argument x is not a vector.");
131 
132  constexpr auto Rows = X::RowsAtCompileTime;
133  constexpr auto MaxRows = X::MaxRowsAtCompileTime;
134 
135  const auto n = x.size();
136  using Gradient = Vec<U, Rows, MaxRows>;
137  Gradient g = Gradient::Zero(n);
138 
139  for(auto i = 0; i < n; ++i)
140  x[i].expr->bind_value(&g[i]);
141 
142  y.expr->propagate(1.0);
143 
144  for(auto i = 0; i < n; ++i)
145  x[i].expr->bind_value(nullptr);
146 
147  return g;
148 }

151 template<typename T, typename X, typename GradientVec>
152 auto hessian(const Variable<T>& y, Eigen::DenseBase<X>& x, GradientVec& g)
153 {
154  using U = VariableValueType<T>;
155 
156  using ScalarX = typename X::Scalar;
157  static_assert(isVariable<ScalarX>, "Argument x is not a vector with Variable<T> (aka var) objects.");
158 
159  using ScalarG = typename GradientVec::Scalar;
160  static_assert(std::is_same_v<U, ScalarG>, "Argument g does not have the same arithmetic type as y.");
161 
162  constexpr auto Rows = X::RowsAtCompileTime;
163  constexpr auto MaxRows = X::MaxRowsAtCompileTime;
164 
165  const auto n = x.size();
166 
167  // Form a vector containing gradient expressions for each variable
168  using ExpressionGradient = Vec<ScalarX, Rows, MaxRows>;
169  ExpressionGradient G(n);
170 
171  for(auto k = 0; k < n; ++k)
172  x[k].expr->bind_expr(&G(k).expr);
173 
174  /* Build a full gradient expression in DFS tree traversal, updating
175  * gradient expressions when encountering variables
176  */
177  y.expr->propagatex(constant<T>(1.0));
178 
179  for(auto k = 0; k < n; ++k) {
180  x[k].expr->bind_expr(nullptr);
181  }
182 
183  // Read the gradient value from gradient expressions' cached values
184  g.resize(n);
185  for(auto i = 0; i < n; ++i)
186  g[i] = val(G[i]);
187 
188  // Form a numeric hessian using the gradient expressions
189  using Hessian = Mat<U, Rows, Rows, MaxRows, MaxRows>;
190  Hessian H = Hessian::Zero(n, n);
191  for(auto i = 0; i < n; ++i)
192  {
193  for(auto k = 0; k < n; ++k)
194  x[k].expr->bind_value(&H(i, k));
195 
196  // Propagate a second derivative value calculation down the gradient expression tree for variable i
197  G[i].expr->propagate(1.0);
198 
199  for(auto k = 0; k < n; ++k)
200  x[k].expr->bind_value(nullptr);
201  }
202 
203  return H;
204 }
