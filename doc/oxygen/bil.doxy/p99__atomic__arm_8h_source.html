282 uint32_t p00_sync_lock_test_and_set_internal(uint32_t volatile *object, memory_order p00_ord) {
283  for (;;) {
284  uint32_t p00_ret = p00_arm_ldrex(object);
285  /* Even if the result has been a 1 in p00_ret, We must imperatively
286  put a strex after the ldex since otherwise we would block other
287  threads when they try to access this. On the other hand even if
288  the strex doesn't succeed but p00_ret is already set, we are also
289  done. */
290  if (!p00_arm_strex(object, 1) || p00_ret) return p00_ret;
291  }
292 }

319 uint32_t __sync_val_compare_and_swap_4(uint32_t volatile *object, uint32_t p00_pre, uint32_t p00_des) {
320  uint32_t p00_ret = 0;
321  for (;;) {
322  p00_ret = p00_arm_ldrex(object);
323  if (p00_pre != p00_ret) {
324  /* wrong value, cancel */
325  p00_arm_strex(object, p00_ret);
326  break;
327  } else {
328  if (!p00_arm_strex(object, p00_des)) break;
329  /* somebody else touched it, continue */
330  }
331  }
332  return p00_ret;
333 }
